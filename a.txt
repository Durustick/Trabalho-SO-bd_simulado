#include <iostream>
#include <pthread.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <cstring>
#include "banco.h"

using namespace std;

Registro banco[MAX];
int total = 0;

pthread_mutex_t mutex;

void* tarefa(void* arg){

    char* comando = (char*)arg; // Recebe o ponteiro unico da thread

    pthread_mutex_lock(&mutex);

    int id;
    char nome[50];

    if(sscanf(comando,"INSERT %d %s",&id,nome)==2){
        bool id_existe = false;
        for(int i = 0; i < total; i++){
            if(banco[i].id == id){
                id_existe = true;
                break;
            }
        }
        if(id_existe){
            cout<<"Erro: Ja existe um registro com o ID "<<id<<"!"<<endl;
        } else if(total < MAX){
            banco[total].id = id;
            strcpy(banco[total].nome,nome);
            total++;
            cout<<"Inserido: "<<id<<" "<<nome<<endl;
        } else {
            cout<<"Erro: Banco de dados cheio!"<<endl;
        }
    }

    else if(sscanf(comando,"SELECT %d",&id)==1){

        for(int i=0;i<total;i++){
            if(banco[i].id==id){
                cout<<"Encontrado: "<<banco[i].id<<" "<<banco[i].nome<<endl;
            }
        }
    }

    else if(sscanf(comando,"DELETE %d",&id)==1){

        for(int i=0;i<total;i++){
            if(banco[i].id==id){
                banco[i]=banco[total-1];
                total--;
                cout<<"Registro removido"<<endl;
            }
        }
    }

    else if(sscanf(comando,"UPDATE %d %s",&id,nome)==2){

        for(int i=0;i<total;i++){
            if(banco[i].id==id){
                strcpy(banco[i].nome,nome);
                cout<<"Atualizado: "<<id<<" "<<nome<<endl;
            }
        }
    }

    pthread_mutex_unlock(&mutex);

    delete[] comando; // Libera a memoria apos o uso

    pthread_exit(NULL);
}

int main(){

    mkfifo("pipe_db",0666);

    // O_RDWR mantem o pipe aberto, evitando loop de EOF quando o cliente sai
    int fd=open("pipe_db",O_RDWR);

    pthread_mutex_init(&mutex,NULL);

    while(true){
        char* comando = new char[100]; // Aloca memoria para evitar data race
        int bytesLidos = read(fd,comando,100);
        
        if(bytesLidos > 0){
            pthread_t thread;
            pthread_create(&thread,NULL,tarefa,comando);
            pthread_detach(thread); // Permite processamento verdadeiramente paralelo
        } else {
            delete[] comando;
        }
    }

    return 0;
}
